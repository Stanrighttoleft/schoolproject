CREATE TABLE shipping_method (
  shipping_id INT AUTO_INCREMENT PRIMARY KEY,
  shipping_name VARCHAR(50) NOT NULL,
  shipping_cost DECIMAL(8,2) NOT NULL,
  shipping_desc VARCHAR(255),
  status TINYINT(1) DEFAULT 1
);
